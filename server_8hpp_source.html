184  void loop(char c) {

186  _ctrl_line.push_back(c);
187  if (std::isalpha(c) || (c == '~')) {
188  if (_ctrl_line.compare("[A") == 0) {

190  _line_index = _line.size() - 1;
191  m_echo_cb(EL_OK, "\r> ", _line, "\033[K");
192  } else if (_ctrl_line.compare("[B") == 0) {
193  _history.next(_line);
194  _line_index = _line.size() - 1;

196  } else if (_ctrl_line.compare("[C") == 0) {
197  if (_line_index < _line.size() - 1) {
198  ++_line_index;
199  m_echo_cb(EL_OK, "\033", _ctrl_line);
200  }
201  } else if (_ctrl_line.compare("[D") == 0) {
202  if (_line_index >= 0) {

204  m_echo_cb(EL_OK, "\033", _ctrl_line);
205  }
206  } else if (_ctrl_line.compare("[H") == 0) {

209  } else if (_ctrl_line.compare("[F") == 0) {

274  el_err_code_t m_exec_cmd(const std::string& cmd) {
275  el_err_code_t ret = EL_EINVAL;
276  std::string cmd_name;
277  std::string cmd_args;
278 
279  size_t pos = cmd.find_first_of("=");
280  if (pos != std::string::npos) {

282  cmd_args = cmd.substr(pos + 1);
283  } else
284  cmd_name = cmd;
285 
286  std::transform(cmd_name.begin(), cmd_name.end(), cmd_name.begin(), ::toupper);
287 
288  if (cmd_name.rfind("AT+", 0) != 0) {
289  m_echo_cb(EL_EINVAL, "Unknown command: ", cmd, "\n");
290  return EL_EINVAL;
291  }
292 
293  cmd_name = cmd_name.substr(3);
294 
295  _cmd_list_lock.lock();
296  auto it = std::find_if(_cmd_list.begin(), _cmd_list.end(), [&](const repl_cmd_t& c) {
297  size_t cmd_body_pos = cmd_name.rfind("@");
298  return c.cmd.compare(cmd_name.substr(cmd_body_pos != std::string::npos ? cmd_body_pos + 1 : 0)) == 0;
299  });
300  if (it == _cmd_list.end()) [[unlikely]] {
301  m_echo_cb(EL_EINVAL, "Unknown command: ", cmd, "\n");
302  _cmd_list_lock.unlock();
303  return ret;
304  }
305  repl_cmd_t cmd_copy = *it;
306  _cmd_list_lock.unlock();
307 
308  if (!cmd_copy.cmd_cb) [[unlikely]]
309  return ret;
310 
311  // tokenize
312  std::vector<std::string> argv;
313  argv.push_back(cmd_name);
314  std::stack<char> stk;
315  size_t index = 0;
316  size_t size = cmd_args.size();
317  while (index < size && argv.size() < (cmd_copy._argc + 1)) {
318  char c = cmd_args.at(index);
319  if (c == '\'' || c == '"') [[unlikely]] {
320  stk.push(c);
321  std::string arg;
322  while (++index < size && stk.size()) {
323  c = cmd_args.at(index);
324  if (c == stk.top())
325  stk.pop();
326  else if (c != '\\') [[likely]]
327  arg += c;
328  else if (++index < size) [[likely]]
329  arg += cmd_args.at(index);
330  }
331  argv.push_back(std::move(arg));
332  } else if (c == '-' || std::isdigit(c)) {
333  size_t prev = index;
334  while (++index < size)
335  if (!std::isdigit(cmd_args.at(index))) break;
336  argv.push_back(cmd_args.substr(prev, index - prev));
337  } else
338  ++index;
339  }
340  argv.shrink_to_fit();
341 
342  if (argv.size() != (cmd_copy._argc + 1)) [[unlikely]] {
343  m_echo_cb(EL_EINVAL, "Command ", cmd_name, " got wrong arguements.\n");
344  return ret;
345  }
346 
347  ret = cmd_copy.cmd_cb(std::move(argv));
348  if (ret != EL_OK) [[unlikely]]
349  m_echo_cb(EL_EINVAL, "Command ", cmd_name, " failed.\n");
350 
351  return ret;
352  }
