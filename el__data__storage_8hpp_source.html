82 static inline unsigned long djb2_hash(const unsigned char* bytes) {
83  unsigned long hash = 0x1505;
84  unsigned char byte;
85  while ((byte = *bytes++)) hash = ((hash << 5) + hash) + byte;
86  return hash;
87 }

91 // currently we're not going to manage the life time of the key chars, and std::type_info is unavailable (due to -fno-rtti)
92 // 1. you can free the buffer using delete[] if you don't need to use the same key anymore (if not, program will crash)
93 // 2. personally I don't think it a good idea to let el_storage_kv_t to manage the lifetime of the key:
94 // a. too compilcate to use const difference to determine whether to delete the key (need to set to nullptr while copy/assign, etc.)
95 // b. not wise to use a bool flag to dertermine whether to delete the key (because copy/assign would be a continuation of the key)
96 template <typename VarType, typename ValueTypeNoCV = typename std::remove_cv<VarType>::type>
97 static inline constexpr el_storage_kv_t<ValueTypeNoCV> el_make_storage_kv_from_type(VarType&& data) {
98  using VarTypeNoCVRef = typename std::remove_cv<typename std::remove_reference<VarType>::type>::type;
99  const char* type_name = get_type_name<VarTypeNoCVRef>();
100  unsigned long hash = djb2_hash(reinterpret_cast<const unsigned char*>(type_name));
101  char* static_buffer = nullptr;
102  static const char* format_str = "edgelab#type_name#%.8lX";
103  static const uint8_t buffer_size = [&]() -> uint8_t {
104  // 1 for terminator, -4 for formatter, 8 for hash hex (unsigned long), -1 for bit hacks (remove last 1 in binary)
105  uint8_t len = static_cast<uint8_t>(std::strlen(format_str) + 1u - 4u + 8u - 1u);
106  len |= len >> 1;
107  len |= len >> 2;
108  len |= len >> 4;
109  return ++len; // the nearest power of 2 value to key length
110  }();
111  static std::forward_list<std::pair<unsigned long, char*>> hash_list{};
112  auto it{std::find_if(hash_list.begin(), hash_list.end(), [&](const std::pair<unsigned long, char*>& pair) {
113  return pair.first == hash;
114  })};
115  if (it != hash_list.end())
116  static_buffer = it->second;
117  else {
118  static_buffer = new char[buffer_size]{}; // we're not going to delete it
119  std::sprintf(static_buffer, format_str, hash);
120  hash_list.emplace_front(hash, static_buffer);
121  }
122  EL_ASSERT(static_buffer != nullptr);
123 
124  return el_make_storage_kv(static_buffer, std::forward<VarType>(data));
125 }

145  // size of the buffer should be equal to handler->value_len
146  template <typename ValueType, typename std::enable_if<!std::is_const<ValueType>::value>::type* = nullptr>
147  bool get(types::el_storage_kv_t<ValueType>& kv) const {
148  const Guard<Mutex> guard(__lock);
149  if (!kv.size || !__kvdb) [[unlikely]]
150  return false;
151 
152  fdb_kv handler{};
153  fdb_kv_t p_handler = fdb_kv_get_obj(__kvdb, kv.key, &handler);
154  if (!p_handler || !p_handler->value_len) [[unlikely]]
155  return false;
156 
157  unsigned char* buffer = new unsigned char[p_handler->value_len];
158  fdb_blob blob{};
159  fdb_blob_read(reinterpret_cast<fdb_db_t>(__kvdb),
160  fdb_kv_to_blob(p_handler, fdb_blob_make(&blob, buffer, p_handler->value_len)));
161  std::memcpy(&kv.value, buffer, std::min(kv.size, static_cast<size_t>(p_handler->value_len)));
162  delete[] buffer;
163 
164  return true;
165  }
